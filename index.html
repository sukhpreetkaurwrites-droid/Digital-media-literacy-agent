(function () { const form = document.getElementById("analyze-form"); const input = document.getElementById("content-input"); const charCount = document.getElementById("char-count"); const analyzeBtn = document.getElementById("analyze-btn"); const errorBox = document.getElementById("error-box"); const loadingBox = document.getElementById("loading-box"); const loadingText = document.getElementById("loading-text"); const results = document.getElementById("results"); const newAnalysisBtn = document.getElementById("new-analysis-btn"); const dateEl = document.getElementById("today-date"); const LOADING_MESSAGES = [ "Cross-referencing manipulation patterns\u2026", "Checking language against known hoax formats\u2026", "Weighing claims against internal consistency\u2026", "Compiling verification steps\u2026", ]; const VERDICT_STYLE = { "Likely Credible": "green", "Needs Verification": "blue", "Likely Misleading": "amber", "Likely False": "red", "Satire": "blue", "Insufficient Information": "amber", }; function today() { const d = new Date(); return d.toLocaleDateString("en-IN", { weekday: "long", year: "numeric", month: "long", day: "numeric" }); } dateEl.textContent = today(); input.addEventListener("input", () => { charCount.textContent = input.value.length; }); function showError(message) { errorBox.textContent = message; errorBox.hidden = false; } function hideError() { errorBox.hidden = true; errorBox.textContent = ""; } function cycleLoadingMessages() { let i = 0; loadingText.textContent = LOADING_MESSAGES[0]; return setInterval(() => { i = (i + 1) % LOADING_MESSAGES.length; loadingText.textContent = LOADING_MESSAGES[i]; }, 1800); } function el(tag, className, text) { const node = document.createElement(tag); if (className) node.className = className; if (text !== undefined) node.textContent = text; return node; } function renderResults(data) { // Verdict stamp const verdictStamp = document.getElementById("verdict-stamp"); const verdictText = document.getElementById("verdict-text"); const styleClass = VERDICT_STYLE[data.verdict] || "amber"; verdictStamp.className = "verdict-stamp " + styleClass; // restart the stamp animation verdictStamp.style.animation = "none"; void verdictStamp.offsetWidth; verdictStamp.style.animation = ""; verdictText.textContent = data.verdict || "Unclear"; // Score gauge const score = Math.max(0, Math.min(100, Number(data.credibility_score) || 0)); document.getElementById("score-number").textContent = score; const needle = document.getElementById("gauge-needle"); requestAnimationFrame(() => { needle.style.left = score + "%"; }); const confidenceChip = document.getElementById("confidence-chip"); confidenceChip.textContent = data.confidence ? data.confidence + " confidence" : ""; // Summary document.getElementById("summary-text").textContent = data.summary || ""; // Red flags const flagsList = document.getElementById("red-flags-list"); const flagsEmpty = document.getElementById("red-flags-empty"); flagsList.innerHTML = ""; const flags = Array.isArray(data.red_flags) ? data.red_flags : []; if (flags.length === 0) { flagsEmpty.hidden = false; } else { flagsEmpty.hidden = true; flags.forEach((flag) => { flagsList.appendChild(el("li", null, flag)); }); } // Manipulation tactics const tacticsGrid = document.getElementById("tactics-grid"); tacticsGrid.innerHTML = ""; const tactics = Array.isArray(data.manipulation_tactics) ? data.manipulation_tactics : []; tactics.forEach((t) => { const card = el("div", "tactic-card" + (t.present_in_content ? " present" : "")); const head = el("div", "tactic-card-head"); head.appendChild(el("span", "tactic-dot")); head.appendChild(el("span", "tactic-name", t.tactic || "Tactic")); card.appendChild(head); if (t.present_in_content) { card.appendChild(el("div", "tactic-present-flag", "Detected in this content")); } card.appendChild(el("p", "tactic-explain", t.explanation || "")); tacticsGrid.appendChild(card); }); // How to verify const verifyList = document.getElementById("verify-list"); verifyList.innerHTML = ""; const steps = Array.isArray(data.how_to_verify) ? data.how_to_verify : []; steps.forEach((step) => verifyList.appendChild(el("li", null, step))); // Sources const chips = document.getElementById("sources-chips"); chips.innerHTML = ""; const sources = Array.isArray(data.trusted_sources_to_check) ? data.trusted_sources_to_check : []; sources.forEach((s) => chips.appendChild(el("span", "source-chip", s))); // Tip document.getElementById("tip-text").textContent = data.media_literacy_tip || ""; results.hidden = false; results.scrollIntoView({ behavior: "smooth", block: "start" }); } form.addEventListener("submit", async (e) => { e.preventDefault(); hideError(); results.hidden = true; const text = input.value.trim(); if (!text) { showError("Please paste some text, a headline, or a claim to analyze."); return; } analyzeBtn.disabled = true; loadingBox.hidden = false; const timer = cycleLoadingMessages(); try { const res = await fetch("/api/analyze", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ text }), }); const data = await res.json(); if (!res.ok) { throw new Error(data.error || "Analysis failed. Please try again."); } renderResults(data); } catch (err) { showError(err.message || "Something went wrong. Please try again."); } finally { clearInterval(timer); loadingBox.hidden = true; analyzeBtn.disabled = false; } }); newAnalysisBtn.addEventListener("click", () => { results.hidden = true; input.value = ""; charCount.textContent = "0"; input.focus(); window.scrollTo({ top: 0, behavior: "smooth" }); }); })();
